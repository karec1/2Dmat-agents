2. Поиск ферромагнетиков с Tc > 100K

Пример на JavaScript
// Получение списка материалов
fetch('{{ request.host_url }}api/materials')
  .then(response => response.json())
  .then(data => {
    console.log('Материалы:', data.materials);
  });

// Поиск ферромагнетиков
fetch('{{ request.host_url }}api/search?magnetic_order=FM')
  .then(response => response.json())
  .then(data => {
    console.log('Ферромагнетики:', data);
  });
